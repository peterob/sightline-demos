-- Core event creation function CREATE OR REPLACE FUNCTION create_audit_event( p_hoa_id UUID, p_event_type VARCHAR(100), p_aggregate_id UUID, p_event_data JSONB, p_signer_user_id UUID ) RETURNS UUID AS $$ DECLARE previous_hash VARCHAR(64); new_event_hash VARCHAR(64); user_cert_fingerprint VARCHAR(128); policy_hash VARCHAR(64); event_id UUID := uuid_generate_v4(); BEGIN -- Get previous event hash for chain linking SELECT event_hash INTO previous_hash FROM audit_events WHERE hoa_id = p_hoa_id ORDER BY sequence_id DESC LIMIT 1; -- Get current policy hash SELECT policy_hash INTO policy_hash FROM policies WHERE hoa_id = p_hoa_id AND superseded_date IS NULL AND policy_type = 'approval_workflow'; -- Get user's certificate fingerprint SELECT certificate_fingerprint INTO user_cert_fingerprint FROM hoa_users WHERE id = p_signer_user_id; -- Calculate event hash new_event_hash := encode( digest( p_event_data::text || policy_hash || COALESCE(previous_hash, '') || NOW()::text, 'sha256' ), 'hex' ); -- Insert immutable event record INSERT INTO audit_events ( hoa_id, event_id, event_type, aggregate_id, event_data, policy_hash, previous_event_hash, event_hash, signer_user_id, signer_cert_fingerprint ) VALUES ( p_hoa_id, event_id, p_event_type, p_aggregate_id, p_event_data, policy_hash, previous_hash, new_event_hash, p_signer_user_id, user_cert_fingerprint ); RETURN event_id; END; $$ LANGUAGE plpgsql; -- Chain verification function CREATE OR REPLACE FUNCTION verify_event_chain(p_hoa_id UUID) RETURNS TABLE(valid BOOLEAN, broken_at BIGINT) AS $$ DECLARE current_event RECORD; expected_hash VARCHAR(64); BEGIN FOR current_event IN SELECT * FROM audit_events WHERE hoa_id = p_hoa_id ORDER BY sequence_id LOOP -- Recalculate hash expected_hash := encode( digest( current_event.event_data::text || current_event.policy_hash || COALESCE(current_event.previous_event_hash, ''), 'sha256' ), 'hex' ); -- Check if hash matches IF current_event.event_hash != expected_hash THEN RETURN QUERY SELECT false, current_event.sequence_id; RETURN; END IF; END LOOP; -- Chain is valid RETURN QUERY SELECT true, NULL::BIGINT; END; $$ LANGUAGE plpgsql;

-- Main verification function CREATE OR REPLACE FUNCTION verify_transaction_complete(p_transaction_id VARCHAR) RETURNS JSONB AS $$ DECLARE verification_result JSONB := '{}'; hash_chain_valid BOOLEAN; signatures_valid BOOLEAN; policy_compliant BOOLEAN; budget_verified BOOLEAN; external_confirmed BOOLEAN; BEGIN -- 1. Verify hash chain integrity SELECT valid INTO hash_chain_valid FROM verify_hash_chain_for_transaction(p_transaction_id); -- 2. Verify all digital signatures SELECT verify_all_signatures(p_transaction_id) INTO signatures_valid; -- 3. Verify policy compliance SELECT verify_policy_compliance(p_transaction_id) INTO policy_compliant; -- 4. Verify budget availability at time of approval SELECT verify_budget_authorization(p_transaction_id) INTO budget_verified; -- 5. Verify external confirmations (bank, etc.) SELECT verify_external_confirmations(p_transaction_id) INTO external_confirmed; -- Build comprehensive result verification_result := jsonb_build_object( 'transaction_id', p_transaction_id, 'verification_timestamp', NOW(), 'overall_status', CASE WHEN hash_chain_valid AND signatures_valid AND policy_compliant AND budget_verified AND external_confirmed THEN 'VERIFIED' ELSE 'FAILED' END, 'hash_chain_integrity', hash_chain_valid, 'digital_signatures', signatures_valid, 'policy_compliance', policy_compliant, 'budget_authorization', budget_verified, 'external_confirmation', external_confirmed, 'verification_time_ms', extract(milliseconds from clock_timestamp() - clock_timestamp()), 'mathematical_certainty', 'IMPOSSIBLE_TO_FORGE' ); RETURN verification_result; END; $$ LANGUAGE plpgsql; -- API endpoint wrapper export async function verifyTransaction(req: Request, res: Response) { const { transaction_id } = req.params; try { // Run complete verification const result = await supabase .rpc('verify_transaction_complete', { p_transaction_id: transaction_id }); // Add human-readable explanations const enrichedResult = { ...result.data, explanations: { hash_chain: "Each transaction links cryptographically to previous ones", signatures: "All approvals verified with RSA-2048 digital signatures", policy: "Automatically checked against immutable policy rules", budget: "Mathematical proof funds were available at approval time", external: "Bank confirmation received and cryptographically verified" }, download_links: { full_audit_trail: `/api/audit-trail/${transaction_id}`, cryptographic_proof: `/api/crypto-proof/${transaction_id}`, policy_documents: `/api/policies/effective-at/${transaction_id}` } }; res.json(enrichedResult); } catch (error) { res.status(500).json({ error: 'Verification failed', details: error.message }); } }
